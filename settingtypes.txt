# all settings for client menu or in minetest.conf on servers

# Enables the wind effects over the airship
steampunk_blimp.enable_wind (enables wind) bool false

# Disables Airship crafting
steampunk_blimp.disable_craftitems (airships cannot be built) bool false

# Enables building the gunboat version of the airship
steampunk_blimp.cannons_enabled (enables building the gunboat version) bool false
